This example does not use standard expand/collapse buttons, instead there are custom buttons which call Expage commands. Also it listens to events

        $(function() {
            $("pre").expage({
                expandButton: $(".doexpand"),
                collapseButton: $(".docollapse"),
                changeHash: false
            }).on("expage.expand", function() {
                $(".events").prepend($("<div>Expanded</div>"));
            }).on("expage.collapse", function() {
                $(".events").prepend($("<div>Collapsed</div>"));
            })
        })
